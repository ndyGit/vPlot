		this.description = 'Requires: [ '+this.accepts.predecessors.join(', ')+' ] Accepts: [ '+this.accepts.successors.join(', ')+' ]';
		/** ********************************** */
		/** PUBLIC METHODS * */
		/** ********************************** */
		/**

					type : config
				}
		 	};
		 }


	});

return plugin;
